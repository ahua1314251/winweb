CREATE TABLE search_spider (
  id bigint unsigned NOT NULL AUTO_INCREMENT COMMENT '主键',
  gmt_create datetime NOT NULL COMMENT '创建时间',
  gmt_modified datetime NOT NULL COMMENT '修改时间',
  gmt_create_origin datetime NOT NULL COMMENT '业务创建时间',
  gmt_modified_origin datetime NOT NULL COMMENT '业务修改时间',
  biz_type varchar(50) NOT NULL COMMENT '垂类类型 其他',
  sub_biz_type varchar(50) NOT NULL COMMENT '垂类二级类型 网站',
  url varchar(300) NOT NULL COMMENT 'web-url地址',
  title text COMMENT '标题',
  img_url text COMMENT '缩略图url',
  content mediumtext NOT NULL COMMENT '索引主内容',
  description mediumtext COMMENT 'desc 摘要',
  keywords mediumtext COMMENT '索引关键字',
  score int NOT NULL DEFAULT '0' COMMENT '索引排序分',
  product_code text COMMENT 'pipcodes 多个英文逗号分隔',
  product_name text COMMENT '产品名称，多个英文逗号分隔',
  nodes_info mediumtext COMMENT '面包屑信息',
  tags text COMMENT '标签英文逗号分隔',
  uk varchar(300) NOT NULL COMMENT '唯一键biz_type@@sub_biz_type@@id',
  out_biz_id varchar(200) NOT NULL COMMENT '爬虫使用url作为唯一主键',
  level tinyint NOT NULL DEFAULT '1' COMMENT '爬虫层级',
  status tinyint NOT NULL DEFAULT '1' COMMENT '1.正常 2.异常',
  version int NOT NULL DEFAULT '0' COMMENT '数据版本控制',
  tag_text text COMMENT '链接来源a标签文本,可能会作为title',
  PRIMARY KEY (id),
  UNIQUE KEY uk_uk (uk(200)),
  KEY idx_biz_type (biz_type),
  KEY idx_sub_biz_type (sub_biz_type),
  KEY idx_out_biz_id (out_biz_id)
) ENGINE=InnoDB AUTO_INCREMENT=127598 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='其他索引内容表';
